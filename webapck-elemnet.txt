 
1.vue init webpack-simple demo 安装脚手架
2.cd demo
3.cnpm i 依赖
4.npm run dev 运行
5.npm i element-ui -S
6.main.js
 引入 :
import Vue from 'vue';
import ElementUI from 'element-ui';
import 'element-ui/lib/theme-chalk/index.css';
import App from './App.vue';
import router from './router'
import vuer from "vue-resource"

Vue.use(ElementUI);
Vue.use(vuer)


new Vue({
  router,
  el: '#app',
  render: h => h(App)
});

7.配置webpack.config.js
  cnpm i style-loader
  webpack.config.js 
  use: [
          'vue-style-loader',
          'style-loader',
          'css-loader'
        ],
 8.npm run dev
 配置webpack.config.js
   test: /\.(png|jpg|gif|svg|ttf|woff|woff2)$/,
9.npm run dev
